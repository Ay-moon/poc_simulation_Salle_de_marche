Responsabilités de COption :

❌ Ne jamais appeler le pricing engine
❌ Ne jamais lire/écrire la base de données
❌ Ne jamais modifier son propre état workflow

C# — Pricing Engine

BlackScholesEngine — Moteur de pricing

C#BlackScholesEngine.cs — DLL COM-visible
using System;
using System.Runtime.InteropServices;

[ComVisible(true)]
[Guid("A1B2C3D4-E5F6-7890-ABCD-EF1234567890")]
[ClassInterface(ClassInterfaceType.None)]
public class BlackScholesEngine : IPricingEngine
{
    // ── Pricing EU option (Call ou Put) ──────────────────────
    public double Price(double S, double K, double r,
                         double sigma, double T, string optType)
    {
        double d1 = (Math.Log(S / K) + (r + sigma * sigma / 2) * T)
                    / (sigma * Math.Sqrt(T));
        double d2 = d1 - sigma * Math.Sqrt(T);

        return optType.ToUpper() == "CALL"
            ? S * N(d1) - K * Math.Exp(-r * T) * N(d2)
            : K * Math.Exp(-r * T) * N(-d2) - S * N(-d1);
    }

    // ── Greeks analytiques (formules fermées) ────────────────
    public double[] Greeks(double S, double K, double r,
                            double sigma, double T, string optType)
    {
        double d1 = (Math.Log(S / K) + (r + sigma * sigma / 2) * T)
                    / (sigma * Math.Sqrt(T));
        double d2 = d1 - sigma * Math.Sqrt(T);
        bool isCall = optType.ToUpper() == "CALL";

        double delta = isCall ? N(d1) : N(d1) - 1.0;
        double gamma = Phi(d1) / (S * sigma * Math.Sqrt(T));
        double vega  = S * Phi(d1) * Math.Sqrt(T) / 100.0; // par 1%
        double theta = isCall
            ? -(S * Phi(d1) * sigma) / (2 * Math.Sqrt(T))
              - r * K * Math.Exp(-r * T) * N(d2)
            : -(S * Phi(d1) * sigma) / (2 * Math.Sqrt(T))
              + r * K * Math.Exp(-r * T) * N(-d2);
        theta /= 365.0; // par jour calendaire
        double rho = isCall
            ? K * T * Math.Exp(-r * T) * N(d2) / 100.0
            : -K * T * Math.Exp(-r * T) * N(-d2) / 100.0;

        return new double[] { delta, gamma, vega, theta, rho };
    }

    // ── Vol implicite par Newton-Raphson ─────────────────────
    public double ImpliedVol(double marketPrice, double S,
                              double K, double r, double T,
                              string optType, int maxIter = 100)
    {
        double sigma = 0.20; // point de départ
        for (int i = 0; i < maxIter; i++)
        {
            double price = Price(S, K, r, sigma, T, optType);
            double vega  = Greeks(S, K, r, sigma, T, optType)[2] * 100;
            if (Math.Abs(vega) < 1e-10) break;
            sigma -= (price - marketPrice) / vega;
            if (Math.Abs(price - marketPrice) < 1e-8) break;
        }
        return sigma;
    }

    // ── Helpers ──────────────────────────────────────────────
    private double N(double x) => /* N cumulatif normal */
        0.5 * (1 + MathNet.Numerics.SpecialFunctions.Erf(x / Math.Sqrt(2)));
    private double Phi(double x) =>
        Math.Exp(-0.5 * x * x) / Math.Sqrt(2 * Math.PI);
}

VolSurface — Interpolation de la surface de vol

VB.NETVolSurface.vb
Public Class VolSurface

    ' Surface stockée : (moneyness, maturity) → impliedVol
    Private _grid As Dictionary(Of (Double, Double), Double)
    Private _moneynessGrid As Double()
    Private _maturityGrid  As Double()

    ''' <summary>Charge la surface depuis un array 2D</summary>
    Public Sub LoadFromArray(data(,) As Object)
        _grid = New Dictionary(Of (Double, Double), Double)
        For i As Integer = 1 To data.GetUpperBound(0)
            Dim m = CDbl(data(i, 1)) ' moneyness = K/F
            Dim t = CDbl(data(i, 2)) ' maturity en années
            Dim v = CDbl(data(i, 3)) ' vol implicite
            _grid((m, t)) = v
        Next
    End Sub

    ''' <summary>Interpolation bilinéaire sur (moneyness, T)</summary>
    Public Function GetVol(spot As Double, strike As Double,
                            T As Double) As Double

        Dim moneyness = strike / spot ' K/S (approx K/F)

        ' Trouver les 4 points encadrants dans la grille
        Dim m_low = FloorMatch(_moneynessGrid, moneyness)
        Dim m_hi  = CeilMatch(_moneynessGrid, moneyness)
        Dim t_low = FloorMatch(_maturityGrid, T)
        Dim t_hi  = CeilMatch(_maturityGrid, T)

        ' Interpolation bilinéaire
        Dim v00 = _grid((m_low, t_low))
        Dim v10 = _grid((m_hi,  t_low))
        Dim v01 = _grid((m_low, t_hi))
        Dim v11 = _grid((m_hi,  t_hi))

        Dim wm = If(m_hi = m_low, 0.5, (moneyness-m_low)/(m_hi-m_low))
        Dim wt = If(t_hi = t_low, 0.5, (T-t_low)/(t_hi-t_low))

        Return (v00 * (1-wm) + v10*wm) * (1-wt) +
               (v01 * (1-wm) + v11*wm) * wt
    End Function

End Class

VaRCalculator — Monte Carlo & VaR historique

Pythonvar_calculator.py — xlwings UDF server
import numpy as np
import pandas as pd
import sqlalchemy as sa
import xlwings as xw

@xw.func
def var_run(book_id: str, confidence: float = 0.99,
            horizon: int = 1, n_simul: int = 10_000) -> dict:
    """
    Calcule VaR 1j 99% pour un book de positions options.
    Appelé depuis VBA via xlwings.call("var_calculator", "var_run", book_id)
    """
    # 1. Charger les positions et Greeks depuis SQL
    engine = sa.create_engine(get_connection_string())
    positions = pd.read_sql(
        f"SELECT * FROM tbl_Positions WHERE BookId = '{book_id}'",
        engine
    )
    greeks = pd.read_sql(
        f"""SELECT p.PositionId, g.Delta, g.Gamma, g.Vega, g.Spot
             FROM tbl_Greeks g JOIN tbl_Positions p ON g.PositionId = p.PositionId
             WHERE p.BookId = '{book_id}' AND g.CalcDate = CAST(GETDATE() AS DATE)""",
        engine
    )

    # 2. Historique 250j de rendements
    returns = pd.read_sql(
        """SELECT UnderlyingId, TradeDate, DailyReturn
           FROM tbl_MarketData WHERE TradeDate >= DATEADD(d,-252,GETDATE())""",
        engine
    ).pivot(index='TradeDate', columns='UnderlyingId', values='DailyReturn')

    # 3. VaR Historique — application des rendements historiques
    hist_pnl = _historical_var(greeks, returns)
    var_hist = np.percentile(hist_pnl, (1 - confidence) * 100)

    # 4. VaR Monte Carlo — simulation GBM corrélée
    sigma = returns.std().values
    corr  = returns.corr().values
    var_mc, es_mc = _monte_carlo_var(greeks, sigma, corr, n_simul, confidence)

    # 5. VaR Paramétrique (delta-normal)
    cov_matrix = np.outer(sigma, sigma) * corr
    delta_vec  = greeks['Delta'].values * greeks['Spot'].values
    port_sigma = np.sqrt(delta_vec @ cov_matrix @ delta_vec)
    var_param  = np.percentile(np.random.normal(0, 1, 100000), (1-confidence)*100) * port_sigma * np.sqrt(horizon)

    return {
        "var_parametric": round(-var_param * _book_notional(positions), 2),
        "var_historical": round(-var_hist, 2),
        "var_mc":         round(-var_mc, 2),
        "es_99":          round(-es_mc, 2),
        "confidence":     confidence,
        "n_simulations":  n_simul
    }


def _monte_carlo_var(greeks, sigma, corr, n_simul, confidence):
    """Monte Carlo VaR : P&L = Δ·ΔS + ½Γ·ΔS²"""
    L = np.linalg.cholesky(corr)
    Z = np.random.randn(n_simul, len(sigma))
    corr_Z   = Z @ L.T
    delta_S  = sigma * corr_Z          # ΔS/S normalisé
    spot_chg = delta_S * greeks['Spot'].values

    pnl = (greeks['Delta'].values * spot_chg
           + 0.5 * greeks['Gamma'].values * spot_chg**2).sum(axis=1)

    var = np.percentile(pnl, (1 - confidence) * 100)
    es  = pnl[pnl <= var].mean()
    return var, es

Option Explicit

Private m_config    As CConfig
Private m_logger    As CLogger
Private m_pricingDll As Object  ' COM : BlackScholesEngine C#
Private m_volSurface As Object  ' COM : VolSurface VB.net
Private m_repo       As CAccessRepository
Private m_workflow   As CWorkflowEngine
Private m_cache      As CPositionCache

Public Sub Initialize()
    On Error GoTo ErrHandler

    ' 1. Infrastructure
    Set m_config = New CConfig
    m_config.LoadFromTable ThisWorkbook.Sheets("Config") _
                            .ListObjects("tbl_Config")

    Set m_logger = New CLogger
    m_logger.Initialize m_config

    ' 2. Pricing Engine COM (.NET DLL enregistrée)
    Set m_pricingDll = CreateObject("CommandoQuant.BlackScholesEngine")
    m_logger.Info "BlackScholesEngine COM loaded ✅"

    ' 3. Vol Surface (VB.net DLL)
    Set m_volSurface = CreateObject("CommandoQuant.VolSurface")
    m_volSurface.LoadFromArray GetVolSurfaceArray()

    ' 4. Data Layer (Access en standalone)
    Set m_repo = New CAccessRepository
    m_repo.Initialize m_config.GetParam("DB_CONNECTION"), m_logger

    ' 5. Cache positions
    Set m_cache = New CPositionCache
    m_cache.Initialize m_repo

    ' 6. Workflow Engine
    Set m_workflow = New CWorkflowEngine
    m_workflow.Initialize m_logger, m_repo, m_pricingDll

    m_logger.Info "CApp initialized — " & m_cache.Count & " positions loaded"
    Exit Sub
ErrHandler:
    MsgBox "FATAL: CommandoQuant init failed." & vbNewLine & Err.Description, vbCritical
End Sub
